void main(void)
{
    int a[21];
    int i;
    int j;
    int k;
    int n;
    i = 0;
    n = 20;
    while( i < 10)
    {
       a[i] = 10 * i ;
       i = i + 1;
     }
    while( i < 21)
    {
        a[i] = 100 - 5 * i;
        i = i + 1;
    }
    i = 0;
    while (i < n)
    {
        j = i + 1;
        while( j < n )
        {
            if(a[j] == a[i])
            {
                k = j;
                while ( k < n)
                {
                    a[k] = a[k+1];
                    k = k + 1;
                }
               n = n - 1 ;
            }
            else
            {
                j = j + 1;
            }
            j = j + 1;
        }
        i = i + 1;
    }
    i = 0;
    while( i < n )
    {
       output(a[i]);
       i = i + 1;
     }
}
